# TROUBLESHOOTING COMMON ISSUES
# ============================================================================
# 
# Issue: "No module named 'tkinter'"
# Fix: sudo apt-get install python3-tk  # Linux only
# 
# Issue: PyTorch installation fails on Apple Silicon
# Fix: pip install --pre torch torchvision torchaudio --index-url https://download.pytorch.org/whl/nightly/cpu
# 
# Issue: geopandas installation fails
# Fix: Install via conda instead: conda install geopandas
# Or: pip install geopandas --no-build-isolation
# 
# Issue: "SSL certificate verify failed" errors
# Fix: pip install --trusted-host pypi.org --trusted-host files.pythonhosted.org -r requirements.txt
# 
# Issue: transformers model download fails
# Fix: Set HuggingFace cache: export HF_HOME=~/.cache/huggingface
# 
# Issue: Memory errors during torch/transformers install
# Fix: Install with --no-cache-dir flag:
#      pip install --no-cache-dir transformers torch
